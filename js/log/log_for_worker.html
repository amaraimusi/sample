ソースコード
	$( function() {
		var logW=new LogForWorker();
		for(var i=0;i<32;i++){
			logW.log(i);
		}
		var fLogs=logW.getFirstLogs();
		$('#res_f').html(fLogs.join(','));
		var lLogs=logW.getLastLogs();
		$('#res_l').html(lLogs.join(','));

	});
