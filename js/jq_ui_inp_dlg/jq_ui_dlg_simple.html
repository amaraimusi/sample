jQuery UI ダイアログ | シンプルひな形

サンプル






	$( function() {
		nekoDlgInit();//ダイアログの初期化
	});
	
	
	/**
	 * ネコダイアログの初期化
	 */
	function nekoDlgInit(){
		//ダイアログの初期化
	    $('#neko_dlg').dialog({
	    	title:'ネコダイアログ',
	        autoOpen: false,
	        width: 400,
	        modal: true,
	    });
	}
	
	/**
	 * ネコダイアログの表示
	 */
	function nekoDlgOpen(){
		$('#neko_dlg').dialog('open');//ダイアログオープン
	
	}
	
	/**
	 * ネコダイアログを閉じる
	 */
	function nekoDlgClose(){
		$('#neko_dlg').dialog('close');
	}
	
	/**
	 * ネコダイアログ、OKボタン押下
	 */
	function nekoDlgOk(){
		//入力フォームからURLを取得する。
		var neko_text = $('#neko_text').val();
	
		//ホスト・ドメイン入力フォームへセットする。
		$('#neko_res').html(neko_text);
		$('#neko_dlg').dialog('close');
	}
	


	<input type="button" value="ダイアログオープン" onclick="nekoDlgOpen()" class="btn btn-primary" />
	<span id="neko_res"><!-- デバッグ出力 --></span>
	
	<!-- URL・ホストドメイン入力ダイアログ -->
	<div id="neko_dlg">
		<input id="neko_text" type="text" class="form-control" placeholder="ネコ名を入力してください" /><br>
		<input type="button" value="OK" onclick="nekoDlgOk()" class="btn btn-success" />
		<input type="button" value="閉じる" onclick="nekoDlgClose()" class="btn btn-primary" />
	</div>
