	// データソース。8bit数値配列形式のバイナリデータ。
	var i8ary = new Uint8Array([137,80,78,71]);
	var source = [ i8ary ];

	// オプション情報。コンテンツタイプ（MIME）など。
	var option = {
		type: "application/octet-binary"
	};

	// blobを作成
	var blob = new Blob( source , option );
