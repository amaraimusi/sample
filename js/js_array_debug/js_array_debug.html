			function debugData(data){
				
				if(data==null){
					debug('データはnullです');
				}else{
					for (var i=0;i<data.length;i++){
						var ent=data[i];
						if(ent==null){
							console.log('エンティティはnull');
						}else{
							msg=ent.join(',');
							console.log(msg);
						}
					}
				}
			}
