スクロール連動

スクロールを連動させるサンプル 
	var div1 = $('#div1');
	var div2 = $('#div2');

	div2.scroll(function() {

		div1.scrollTop(div2.scrollTop());//縦スクロールを連動させる。
		div1.scrollLeft(div2.scrollLeft());//横スクロールを連動させる。

	});
