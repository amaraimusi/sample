UNIXタイムスタンプと日時を相互変換

検証1:jquery版

var u1 = Math.floor(new Date().getTime() / 1000);
$('#res1').html(u1);

検証2:jquery版

var u2 = Math.floor($.now() / 1000);
$('#res2').html(u2);

検証3

var d3 = new Date();
$('#res3').append('デフォルト日時 → ' + d3 + '<br>');
var d3_ymd = d3.toLocaleString();
$('#res3').append('YMD日時 → ' + d3_ymd + '<br>');
var d3_u = Math.floor(d3);
$('#res3').append('UNITタイムスタンプ → ' + d3_u + '<br>');
var d3_rev = new Date(d3_u);
$('#res3').append('UNITタイムスタンプを日時に戻す  → ' + d3_rev + '<br>');
