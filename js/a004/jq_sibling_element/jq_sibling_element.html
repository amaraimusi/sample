TEST1:

自分以外の兄弟要素を取得する。

	var elms = $('#test1').siblings();
	elms.each(function(){
		var str = $(this).html() + '<br>';
		$('#res').append(str);
	});
	







	
TEST2:

	条件を指定して兄弟要素を取得する。

	下記の例では条件にDIV要素を指定している。
	

	var elms = $('#test1').siblings('div');
	elms.each(function(){
		var str = $(this).html() + '<br>';
		$('#res2').append(str);
	});
	

	
	










	
TEST3:

	自分から下の弟要素だけ取得する。（自分自身は含めず）
	

	function test3(){
		var elms = $('#test1').nextAll();
		elms.each(function(){
			var str = $(this).html() + '<br>';
			$('#res3').append(str);
		});
	}
	

	
	








	
TEST4:

	自分自身も含めてすべての兄弟要素を取得する。
	

	function test4(){
		var elms = $('#test1').parent().children();
		elms.each(function(){
			var str = $(this).html() + '<br>';
			$('#res4').append(str);
		});
	}
