	"require": {
		"tecnickcom/tcpdf": "*"
	}
}

HTML 

<a href="demo1_pdf.php" class="btn btn-success">PDFを表示</a>

PHP(demo1_pdf.php) 

<?php
include '../../../composer/vendor/autoload.php';
$tcpdf = new TCPDF("Portrait");
$tcpdf->AddPage();
$tcpdf->SetFont("kozminproregular", "", 10);
$html = <<< EOF
<style>
	h1 {
		font-weight:bold;
		font-size: 24px;
		color:#FF8080;
	}

</style>
<h1>Hello World!</h1>
いろはにほへと<br>
大きなヤギと小さな犬<br>
EOF;

$tcpdf->writeHTML($html);
$tcpdf->Output("test.pdf");
